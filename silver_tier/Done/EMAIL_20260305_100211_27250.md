The rest is kinda boilerplate.. initialize agent, prefetch some data etc. which happens in the context of a asyncio runner bootstrapped to a VoIP call context. But the salient parts are these



```

    def _setup_agent_session(self, parent_ctx):

        """Run the realtime-agent session in its own asyncio loop."""

        with propagate.attach_context(parent_ctx):



            asyncio.set_event_loop(self._agent_loop)

            session_ctx = self._agent_loop.run_until_complete(self.runner.run())

            self.session = self._agent_loop.run_until_complete(session_ctx.__aenter__())



            # start background task to harvest events

            self._agent_loop.create_task(self._process_agent_events(parent_ctx))

            # Send an initial greeting to the caller as soon as the session is ready

            asyncio.run_coroutine_threadsafe(

                self.session.send_message("Greet the caller and ask how you can help!"),

                self._agent_loop

            )

            self._agent_loop.run_forever()

```
