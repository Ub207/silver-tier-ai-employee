Also encountering this. Chat suggested this workaround and it seems to work for now until this is fixed (I’d like/expect to have tools/thinking in the history since it’s server-side w/ this session):

```python

from agents import Session, TResponseInputItem



class SafeConversations(Session):

    def __init__(self, inner: Session):

        self.inner = inner



    async def get_items(self, limit=None):

        return await self.inner.get_items(limit=limit)



    async def add_items(self, items: list[TResponseInputItem]) -> None:

        cleaned = []

        for it in items:

            if it.get("type") == "message" and it.get("role") in ("user", "assistant"):

                parts = it.get("content") or []

                text_parts = [p for p in parts if p.get("type") == "text" and isinstance(p.get("text"), str)]

                if text_parts:

                    cleaned.append({"type": "message", "role": it["role"], "content": text_parts})

        if cleaned:

            await self.inner.add_items(cleaned)



    async def pop_item(self):

        return await self.inner.pop_item()



from agents import OpenAIConversationsSession

base = OpenAIConversationsSession()

session = SafeConversations(base)

```
