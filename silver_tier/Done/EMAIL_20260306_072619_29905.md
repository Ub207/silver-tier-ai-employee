```python

import asyncio



from agents import Agent

from agents.extensions.models.litellm_model import LitellmModel

from agents.model_settings import ModelSettings

from agents.run import Runner

from agents.tool import function_tool

from openai.types import Reasoning





def get_weather(location: str) -> str:

    """Get the weather for a location."""

    return f"The weather in {location} is Sunny with a temperature of 22C."





async def main():

    agent = Agent(

        name="weather_assistant",

        model=LitellmModel("anthropic/claude-sonnet-4-20250514"),

        instructions="You are a helpful assistant.",

        tools=[function_tool(get_weather)],

        model_settings=ModelSettings(

            include_usage=True,

            reasoning=Reasoning(effort="medium"),  # Enable reasoning

        ),

    )



    response1 = await Runner().run(

        agent, "Hi! Tell me a joke FIRST and then what's the weather like in Tokyo?"

    )

    print(response1.final_output)





if __name__ == "__main__":

    asyncio.run(main())

```
