import asyncio

from agents import Agent, ItemHelpers, ModelSettings, Runner, function_tool



@function_tool

def foobar() -> int:

    return 10



async def main():

    agent = Agent(

        name="foobar",

        model="gpt-5-mini",

        instructions="First think and call the `foobar` tool, then just tell me the number, only number",

        tools=[foobar],

        model_settings=ModelSettings( reasoning={ "effort": "high", "summary": "auto" }),        

    )



    result = Runner.run_streamed(agent, "Hello")



    async for event in result.stream_events():

        # Handle raw response events with progress indicators

        if event.type == "raw_response_event":

            item = getattr(event.data, "item", None)

            if item:

                print(f"[raw_response_event] Type: {event.data.type} ItemType: {item.type}")

            else:

                print(f"[raw_response_event] Type: {event.data.type}" )

                                           

        elif event.type == "agent_updated_stream_event":

            print(f"[agent_updated_stream_event]: {event.new_agent.name}")

        elif event.type == "run_item_stream_event":

            print(f"[run_item_stream_event] {event.item.type}")



if __name__ == "__main__":

    asyncio.run(main())
