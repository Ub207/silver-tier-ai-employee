@dannyhorcajo Actually, I found that regular @function_tool do support the `is_enabled` parameter — so it’s possible to dynamically enable or disable tools based on the RunContextWrapper data.



Here’s a simple example: the bar tool will only be available after the foo tool has been invoked.



```python

from dataclasses import dataclass

from agents import Agent, function_tool, Runner,RunContextWrapper, AgentBase



def is_bar_enabled(ctx: RunContextWrapper, agent: AgentBase) -> bool:

    """Check if bar tool should be enabled based on whether foo was called."""

    return ctx.context.executed_tools.get("foo", False)



@function_tool

async def foo(ctx: RunContextWrapper) -> str:

    ctx.context.executed_tools["foo"] = True

    return "41"



@function_tool(is_enabled=is_bar_enabled)

async def bar() -> str:

    return "64"



@dataclass

class RunContext:

    executed_tools: dict[str, bool]



agent = Agent[RunContext](

    name="Assistant",

    instructions="You are a helpful assistant. 用中文回答.",

    model="gpt-4.1-mini",

    tools=[foo, bar]

)



async def main():

    """Main function demonstrating conditional tool availability."""

    run_context = RunContext(executed_tools={})



    print("-------------------------------- Round 1 --------------------------------")



    messages = [{"role": "user", "content": "Call the foo tool first. After getting the result, you can call the bar tool, and finally tell me the result."}]



    result = await Runner.run(agent, input=messages, context=run_context)



    print(result.final_output)

    

    print("-------------------------------- Round 2 --------------------------------")

    

    messages = result.to_input_list()



    messages.append({"role": "user", "content": "Call the bar tool again and tell me the result"})

    result = await Runner.run(agent, input=messages, context=run_context)
