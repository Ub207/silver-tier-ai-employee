Here is a minimal example:

```python

#!/usr/bin/env python3

from openai import AsyncOpenAI

from agents import Agent, Runner, set_default_openai_client, set_tracing_disabled, set_default_openai_api, function_tool

import os



set_tracing_disabled(True)

set_default_openai_api('chat_completions')



@function_tool

def get_weather(city: str) -> str:

    """returns weather info for the specified city."""

    return f"The weather in {city} is sunny"



_client = AsyncOpenAI(base_url=os.getenv("OPENAI_URL")) # This is a GPUStack server wrapping vLLM v0.10.1.1

set_default_openai_client(_client)



fc_agent = Agent(name="fc_agent", model="qwen3-32b", tools=[get_weather])



result = Runner.run_sync(fc_agent, "What's the weather like in Seattle?")



print(f"Intermediate output: {result.to_input_list()}")



haiku_agent = Agent(

        name="Haiku agent",

        instructions="Always respond in haiku form",

        model="gpt-oss-120b"

        )



result = Runner.run_sync(haiku_agent, result.to_input_list())

print(result.final_output)

```
